    """
    Manager for surfaces used in photon tracking simulations.

    This class initializes and manages surface properties for simulation purposes,
    including creating different types of surfaces and handling their optical properties.

    Attributes:
        surface_data_path (str): Path to the CSV file containing surface properties data.
        SiPMAOIref_path (str): Path to the CSV file containing SiPM AOI reflectivity data.
        run_id (int): ID of the current simulation run.
        mat_manager (material_manager): Instance of material_manager for managing material properties.
        wavelengths (list): List of standard wavelengths used in simulations.
        num_wavelengths (int): Number of wavelengths defined in self.wavelengths.
        surfaces (dict): Dictionary mapping surface names to Surface objects.
        surface_props (dict): Dictionary storing surface properties read from the CSV file.

    Methods:
        __init__(self, material_manager, experiment_name, run_id):
            Initialize the surface manager with the material manager, experiment name, and run ID.
            Reads surface data and builds Surface objects.

        build_surfaces(self):
            Build Surface objects from the surface data read from the CSV file.
            Initializes surface properties and stores them in self.surfaces and self.surface_props.

        get_surface(self, surface_name):
            Retrieve a Surface object by its name.

        create_dielectric_metal_surface(self, name, inner_mat):
            Create a dielectric-metal surface with specified inner material properties.

        SiPMEmpirical_surface(self, name):
            Create a SiPM empirical surface with reflectivity properties.

        create_dichroic_surface(self, name, inner_mat, outer_mat):
            Create a dichroic surface with specified inner and outer material properties.

        calc_R_T(self, n1, eta2, k2, num_angles):
            Calculate reflectance (R) and transmittance (T) for a given incidence angle range.

        get_eta2_k2(self, eta2, k2):
            Store and return eta2 and k2 values for the current instance.
    """

[docs]    def get_surface(self, surface_name):
        """
        Retrieve a Surface object by its name.

        Args:
            surface_name (str): Name of the surface to retrieve.

        Returns:
            Surface: Surface object corresponding to the given name.

        Raises:
            Exception: If the surface name does not exist in self.surfaces.
        """
        if surface_name == 'None':
            return None
        elif surface_name in self.surfaces:
            return self.surfaces[surface_name]
        else:
            raise Exception('Surface does not exist: ' + surface_name)

        """
        Create a dielectric-metal surface with specified inner material properties.

        Args:
            name (str): Name of the surface.
            inner_mat (str): Name of the inner material.

        Returns:
            Surface: Dielectric-metal Surface object.
        """
        dielectric_metal_surface = Surface(name, model=4)
        eta2 = float(self.mat_manager.material_props[inner_mat]['eta'])
        k2 = float(self.mat_manager.material_props[inner_mat]['k'])
        dielectric_metal_surface.set('eta', eta2)
        dielectric_metal_surface.set('k', k2)

        return dielectric_metal_surface

        """
        Create a SiPM empirical surface with reflectivity properties.

        Args:
            name (str): Name of the surface.

        Returns:
            Surface: SiPM empirical Surface object.
        """
        # print('model 5')

        sipmEmpirical_surface = Surface(name, model=5)

        self.SiPMreflctivity = pd.read_csv(self.SiPMAOIref_path)
        # num_angle = 90

        # inci_theta_deg = np.linspace(0,90,num_angle,dtype = np.float32)
        # inci_theta = np.radians(inci_theta_deg)

        ref = []
        rela_PDE = []
        inci_theta_csv = self.SiPMreflctivity['AOI'].tolist()
        inci_theta = np.radians(inci_theta_csv)
        ref_csv = self.SiPMreflctivity['reflectivity'].tolist()

        for i in range(0, len(inci_theta_csv)):
            r_SiPM = np.zeros((self.num_wavelengths, 2), dtype=np.float32)
            r_SiPM[:, 0] = self.wavelengths
            # r_SiPM[:, 1] = random.random()
            r_SiPM[:, 1] = ref_csv[i]
            # print(r_SiPM[:,1])
            ref.append(r_SiPM)
            relative_PDE = np.zeros((self.num_wavelengths, 2), dtype=np.float32)
            relative_PDE[:, 0] = self.wavelengths
            relative_PDE[:, 1] = 1.0
            rela_PDE.append(relative_PDE)
        SiPM_props = SiPMEmpiricalProps(inci_theta, ref, rela_PDE)
        sipmEmpirical_surface.sipmEmpirical_props = SiPM_props
        # print(sipmEmpirical_surface.sipmEmpirical_props)
        return sipmEmpirical_surface

[docs]    def calc_R_T(self, n1, eta2, k2, num_angles):
        """
        Calculate reflectance (R) and transmittance (T) for a given incidence angle range.

        Args:
            n1 (float): Refractive index of the outer material.
            eta2 (float): Real part of the complex refractive index of the inner material.
            k2 (float): Imaginary part of the complex refractive index of the inner material.
            num_angles (int): Number of angles to calculate.

        Returns:
            tuple: Tuple containing theta (angle), reflectance (R), and transmittance (T).
        """
        theta = np.linspace(0, np.pi / 2, num_angles)
        R = np.zeros_like(theta)
        T = np.zeros_like(theta)

        for i in range(num_angles):
            r, t = chroma.geometry.fresnel_coefficients(n1, eta2, k2, theta[i])
            R[i] = r
            T[i] = t

        return theta, R, T
