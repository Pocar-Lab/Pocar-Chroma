class material_manager:
    """
    Manager for materials used in photon tracking simulations.

    This class reads material data from a CSV file, initializes Material objects,
    and manages material properties for simulation purposes.

    Attributes:
        material_data_path (str): Path to the CSV file containing material data.
        run_id (int): ID of the current simulation run.
        materials (dict): Dictionary mapping material names to Material objects.
        material_props (dict): Dictionary storing material properties read from the CSV file.

    Methods:
        __init__(self, experiment_name, run_id):
            Initialize the material manager with the experiment name and run ID.
            Reads material data and builds Material objects.

        add_attributes(self, curr_material, refractive_index=None, absorption_length=None,
                       scattering_length=None, density=0.0):
            Add optical properties and density to a Material object.

        build_materials(self, run_id):
            Build Material objects from the material data read from the CSV file.
            Initializes material properties and stores them in self.materials and self.material_props.

        get_material(self, material_name):
            Retrieve a Material object by its name.

    Notes:
        - This class relies on the 'Material' class from the chroma.geometry module.
        - Material properties are initialized from the CSV file and can be modified or updated during simulation.
        - The CSV file should contain columns for 'name', 'refractive_index', 'absorption_length', 'scattering_length',
          and 'density' for each material.
    """

    def add_attributes(self, curr_material, refractive_index=None, absorption_length=None,
                       scattering_length=None, density=0.0):
        """
        Add optical properties and density to a Material object.

        Args:
            curr_material (Material): Material object to which properties are added.
            refractive_index (float or None): Refractive index of the material.
            absorption_length (float or None): Absorption length of the material.
            scattering_length (float or None): Scattering length of the material.
            density (float): Density of the material.

        Returns:
            None
        """
        curr_material.set('refractive_index', refractive_index)
        curr_material.set('absorption_length', absorption_length)
        curr_material.set('scattering_length', scattering_length)
        curr_material.density = density

    def build_materials(self, run_id):
        """
        Build Material objects from material data read from the CSV file.

        Args:
            run_id (int): ID of the current simulation run.

        Returns:
            None
        """
        self.materials_df = pd.read_csv(self.material_data_path)
        self.materials = {}
        self.material_props = {}

        for index, row in self.materials_df.iterrows():
            curr_name = row['name']
            self.materials[curr_name] = Material(name=curr_name)
            self.add_attributes(self.materials[curr_name],
                                refractive_index=row['refractive_index'],
                                absorption_length=row['absorption_length'],
                                scattering_length=row['scattering_length'],
                                density=row['density'])
            self.material_props[curr_name] = dict(row)

    def get_material(self, material_name):
        """
        Retrieve a Material object by its name.

        Args:
            material_name (str): Name of the material to retrieve.

        Returns:
            Material: Material object corresponding to the given name.

        Raises:
            Exception: If the material name does not exist in self.materials.
        """
        if material_name in self.materials:
            return self.materials[material_name]
        else:
            raise Exception('Material does not exist: ' + material_name)
